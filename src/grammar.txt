<expr>          ::= var <identifier> = <expr>
                ::= <compare-expr> ((and | or) <compare-expr>)*
                
<compare-expr>  ::= (not) <compare-expr>
                ::= <arith-expr> (('==' | '!=' | '<' | '<=' | '>' | '>) <arith-expr>)*

<arith-expr>    ::= <term> (('+' | '-') <term>)*

<term>          ::= <factor> (('*' | '/') <factor>)*

<factor>        ::= call (^ <factor>)*

<call>          ::= atom (<expr> (',' <expr>)* )?

<atom>          ::= INT/LONGINT/FLOAT/DOUBLE
                ::= ('+' | '-') <atom>
                ::= ( <expr> )
                ::= <if-expr>
                ::= <func-def>

<if-expr>       ::= if <expr> { <expr> }
                        (elif { <expr> })*
                        (else { <expr> })?

<func-def>      ::= func <identifier>( (<identifier> (',' <identifier>)* )? ) { expr }
